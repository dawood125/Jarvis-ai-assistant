Implementing a JWT flow with Synthetix Core requires initializing the AuthModule. Here is the standard implementation for your controller:

async function initializeSecureSession(user_id) {
const core = await Synthetix.connect();
const payload = {
sub: user_id,
iat: Date.now(),
scope: ['read:vault', 'write:terminal']
};
// Sign the token using the internal HSM module
return core.auth.sign(payload, { algorithm: 'RS256' });
}
